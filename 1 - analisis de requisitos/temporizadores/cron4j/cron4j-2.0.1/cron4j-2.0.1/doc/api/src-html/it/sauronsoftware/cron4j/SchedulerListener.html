019    package it.sauronsoftware.cron4j;
020    
021    /**
022     * <p>
023     * Implement this interface and register your instance with the
024     * {@link Scheduler#addSchedulerListener(SchedulerListener)} method to receive
025     * notifications about scheduled task executions.
026     * </p>
027     * 
028     * @author Carlo Pelliccia
029     * @since 2.0
030     */
031    public interface SchedulerListener {
032    
033            /**
034             * This one is called by the scheduler when a task execution is starting.
035             * 
036             * @param executor
037             *            The task executor.
038             */
039            public void taskLaunching(TaskExecutor executor);
040    
041            /**
042             * This one is called by the scheduler to notify that a task execution has
043             * been successfully completed.
044             * 
045             * @param executor
046             *            The task executor.
047             */
048            public void taskSucceeded(TaskExecutor executor);
049    
050            /**
051             * This one is called by the scheduler to notify that a task execution has
052             * failed.
053             * 
054             * @param executor
055             *            The task executor.
056             * @param exception
057             *            The exception representing the failure notification.
058             */
059            public void taskFailed(TaskExecutor executor, Throwable exception);
060    
061    }
