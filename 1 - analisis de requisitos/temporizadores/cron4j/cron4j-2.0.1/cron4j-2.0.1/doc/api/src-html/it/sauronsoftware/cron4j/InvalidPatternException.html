021    /**
022     * <p>
023     * This kind of exception is thrown if an invalid scheduling pattern is
024     * encountered by the scheduler.
025     * </p>
026     * 
027     * @author Carlo Pelliccia
028     */
029    public class InvalidPatternException extends RuntimeException {
030    
031            private static final long serialVersionUID = 1L;
032    
033            /**
034             * Package-reserved construction.
035             */
036            InvalidPatternException() {
037            }
038    
039            /**
040             * Package-reserved construction.
041             */
042            InvalidPatternException(String message) {
043                    super(message);
044            }
045    
046    }
