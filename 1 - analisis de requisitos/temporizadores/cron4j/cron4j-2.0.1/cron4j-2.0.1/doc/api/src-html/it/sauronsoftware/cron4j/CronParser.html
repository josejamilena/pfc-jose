032    /**
033     * <p>
034     * A parser for crontab-like formatted files and streams.
035     * </p>
036     * <p>
037     * If you want to schedule a list of tasks declared in a crontab-like file you
038     * don't need the CronParser, since you can do it by adding the file to the
039     * scheduler, with the {@link Scheduler#scheduleFile(File)} method.
040     * </p>
041     * <p>
042     * Consider to use the CronParser if the {@link Scheduler#scheduleFile(File)}
043     * method is not enough for you. In example, you may need to fetch the task list
044     * from a remote source which is not representable as a {@link File} object (a
045     * document on a remote server, a DBMS result set and so on). To solve the
046     * problem you can implement your own {@link TaskCollector}, getting the
047     * advantage of the CronParser to parse easily any crontab-like content.
048     * </p>
049     * <p>
050     * You can parse a whole file/stream, but you can also parse a single line.
051     * </p>
052     * <p>
053     * A line can be empty, can contain a comment or it can be a scheduling line.
054     * </p>
055     * <p>
056     * A line containing no characters or a line with only space characters is
057     * considered an empty line.
058     * </p>
059     * <p>
060     * A line whose first non-space character is a number sign (#) is considered a
061     * comment.
062     * </p>
063     * <p>
064     * Empty lines and comment lines are ignored by the parser.
065     * </p>
066     * <p>
067     * Any other kind of line is parsed as a scheduling line.
068     * </p>
069     * <p>
070     * A valid scheduling line respects the following structure:
071     * </p>
072     * 
073     * <pre>
074     * scheduling-pattern [options] command [args]
075     * </pre>
076     * 
077     * <ul>
078     * <li><em>scheduling-pattern</em> is a valid scheduling pattern, according with
079     * the definition given by the {@link SchedulingPattern} class.</li>
080     * <li><em>options</em> is a list of optional informations used by cron4j to
081     * prepare the task execution environment. See below for a more detailed
082     * description.</li>
083     * <li><em>command</em> is a system valid command, such an executable call.</li>
084     * <li><em>args</em> is a list of optional arguments for the command.</li>
085     * </ul>
086     * <p>
087     * After the scheduling pattern item, other tokens in each line are space
088     * separated or delimited with double quotation marks (&quot;).
089     * </p>
090     * <p>
091     * Double quotation marks delimited items can take advantage of the following
092     * escape sequences:
093     * </p>
094     * <ul>
095     * <li>\&quot; - quotation mark</li>
096     * <li>\\ - back slash</li>
097     * <li>\/ - slash</li>
098     * <li>\b - back space</li>
099     * <li>\f - form feed</li>
100     * <li>\n - new line</li>
101     * <li>\r - carriage return</li>
102     * <li>\t - horizontal tab</li>
103     * <li>\u005c\u0075<em>four-hex-digits</em> - the character at the given unicode
104     * index</li>
105     * </ul>
106     * <p>
107     * The <em>options</em> token collection can include one or more of the
108     * following elements:
109     * </p>
110     * <ul>
111     * <li>IN:<em>file-path</em> - Redirects the command standard input channel to
112     * the specified file.</li>
113     * <li>OUT:<em>file-path</em> - Redirects the command standard output channel to
114     * the specified file.</li>
115     * <li>ERR:<em>file-path</em> - Redirects the command standard error channel to
116     * the specified file.</li>
117     * <li>ENV:<em>name</em>=<em>value</em> - Defines an environment variable in the
118     * scope of the command.</li>
119     * <li>DIR:<em>directory-path</em> - Sets the path of the working directory for
120     * the command. This feature is not supported if the executing JVM is less than
121     * 1.3.</li>
122     * </ul>
123     * <p>
124     * Invalid scheduling lines are discarded without blocking the parsing
125     * procedure, but an error message is printed in the application standard error
126     * channel.
127     * </p>
128     * <p>
129     * Valid examples:
130     * </p>
131     * 
132     * <pre>
133     * 0 5 * * * sol.exe
134     * 0,30 * * * * OUT:C:\ping.txt ping 10.9.43.55
135     * 0,30 4 * * * &quot;OUT:C:\Documents and Settings\Carlo\ping.txt&quot; ping 10.9.43.55
136     * 0 3 * * * ENV:JAVA_HOME=C:\jdks\1.4.2_15 DIR:C:\myproject OUT:C:\myproject\build.log C:\myproject\build.bat &quot;Nightly Build&quot;
137     * </pre>
138     * 
139     * @author Carlo Pelliccia
140     * @since 2.0
141     */
142    public class CronParser {

145             * Instantiation prohibited.
146             */
147            private CronParser() {

150            /**
151             * <p>
152             * Builds a task list reading it from a file.
153             * </p>
154             * 
155             * <p>
156             * The file is treated as UTF-8. If your source file is not UTF-8 encoded
157             * establish by yourself a {@link Reader} using the right charset and pass
158             * it to the {@link CronParser#parse(Reader)} method.
159             * </p>
160             * 
161             * <p>
162             * Syntax and semantics errors in the source file are not blocking. Invalid
163             * lines are discarded, and they cause just a stack trace to be printed in
164             * the standard error channel as a notification.
165             * </p>
166             * 
167             * @param file
168             *            The file.
169             * @return The task table parsed from the file.
170             * @throws IOException
171             *             I/O error.
172             */
173            public static TaskTable parse(File file) throws IOException {
174                    InputStream stream = null;
175                    try {
176                            stream = new FileInputStream(file);
177                            return parse(stream);
178                    } finally {
179                            if (stream != null) {
180                                    try {
181                                            stream.close();
182                                    } catch (Throwable t) {
183                                            ;
184                                    }
185                            }
186                    }
187            }

189            /**
190             * <p>
191             * Builds a task list reading it from an URL.
192             * </p>
193             * 
194             * <p>
195             * Contents fetched from the URL are treated as UTF-8. If your source is not
196             * UTF-8 encoded establish by yourself a {@link Reader} using the right
197             * charset and pass it to the {@link CronParser#parse(Reader)} method.
198             * </p>
199             * 
200             * <p>
201             * Syntax and semantics errors in the retrieved document are not blocking.
202             * Invalid lines are discarded, and they cause just a stack trace to be
203             * printed in the standard error channel as a notification.
204             * </p>
205             * 
206             * @param url
207             *            The URL.
208             * @return The task table parsed from the contents fetched from the given
209             *         URL.
210             * @throws IOException
211             *             I/O error.
212             */
213            public static TaskTable parse(URL url) throws IOException {
214                    InputStream stream = null;
215                    try {
216                            stream = url.openStream();
217                            return parse(stream);
218                    } finally {
219                            if (stream != null) {
220                                    try {
221                                            stream.close();
222                                    } catch (Throwable t) {
223                                            ;
224                                    }
225                            }
226                    }
227            }

229            /**
230             * <p>
231             * Builds a task list reading it from an input stream.
232             * </p>
233             * 
234             * <p>
235             * The stream is treated as UTF-8. If your source is not UTF-8 encoded
236             * establish by yourself a {@link Reader} using the right charset and pass
237             * it to the {@link CronParser#parse(Reader)} method.
238             * </p>
239             * 
240             * <p>
241             * Syntax and semantics errors in the source stream are not blocking.
242             * Invalid lines are discarded, and they cause just a stack trace to be
243             * printed in the standard error channel as a notification.
244             * </p>
245             * 
246             * @param stream
247             *            The input stream.
248             * @return The task table parsed from the stream contents.
249             * @throws IOException
250             *             I/O error.
251             */
252            public static TaskTable parse(InputStream stream) throws IOException {
253                    return parse(new InputStreamReader(stream, "UTF-8"));
254            }
255    
256            /**
257             * <p>
258             * Builds a task list reading it from a reader.
259             * </p>
260             * 
261             * <p>
262             * Syntax and semantics errors in the source reader are not blocking.
263             * Invalid lines are discarded, and they cause just a stack trace to be
264             * printed in the standard error channel as a notification.
265             * </p>
266             * 
267             * @param reader
268             *            The reader.
269             * @return The task table parsed from the contents in the reader.
270             * @throws IOException
271             *             I/O error.
272             */
273            public static TaskTable parse(Reader reader) throws IOException {
274                    TaskTable table = new TaskTable();
275                    BufferedReader bufferedReader = new BufferedReader(reader);
276                    try {
277                            String line;
278                            while ((line = bufferedReader.readLine()) != null) {
279                                    try {
280                                            parseLine(table, line);
281                                    } catch (Exception e) {
282                                            e.printStackTrace();
283                                            continue;
284                                    }
285                            }
286                    } finally {
287                            reader.close();
288                    }
289                    return table;
290            }
291    
292            /**
293             * Parses a crontab-like line.
294             * 
295             * @param table
296             *            The table on which the parsed task will be stored, by
297             *            side-effect.
298             * @param line
299             *            The crontab-like line.
300             * @throws Exception
301             *             The supplied line doesn't represent a valid task line.
302             */
303            public static void parseLine(TaskTable table, String line) throws Exception {
304                    line = line.trim();
305                    if (line.length() == 0 || line.charAt(0) == '#') {
306                            return;
307                    }
308                    // Detecting the pattern.
309                    int size = line.length();
310                    String pattern = null;
311                    for (int i = size; i >= 0; i--) {
312                            String aux = line.substring(0, i);
313                            if (SchedulingPattern.validate(aux)) {
314                                    pattern = aux;
315                                    break;
316                            }
317                    }
318                    if (pattern == null) {
319                            throw new Exception("Invalid cron line: " + line);
320                    }
321                    line = line.substring(pattern.length());
322                    size = line.length();
323                    // Splitting the line
324                    ArrayList splitted = new ArrayList();
325                    StringBuffer current = null;
326                    boolean quotes = false;
327                    for (int i = 0; i < size; i++) {
328                            char c = line.charAt(i);
329                            if (current == null) {
330                                    if (c == '"') {
331                                            current = new StringBuffer();
332                                            quotes = true;
333                                    } else if (c > ' ') {
334                                            current = new StringBuffer();
335                                            current.append(c);
336                                            quotes = false;
337                                    }
338                            } else {
339                                    boolean closeCurrent;
340                                    if (quotes) {
341                                            closeCurrent = (c == '"');
342                                    } else {
343                                            closeCurrent = (c <= ' ');
344                                    }
345                                    if (closeCurrent) {
346                                            if (current != null && current.length() > 0) {
347                                                    String str = current.toString();
348                                                    if (quotes) {
349                                                            str = escape(str);
350                                                    }
351                                                    splitted.add(str);
352                                            }
353                                            current = null;
354                                    } else {
355                                            current.append(c);
356                                    }
357                            }
358                    }
359                    if (current != null && current.length() > 0) {
360                            String str = current.toString();
361                            if (quotes) {
362                                    str = escape(str);
363                            }
364                            splitted.add(str);
365                            current = null;
366                    }
367                    // Analyzing
368                    size = splitted.size();
369                    int status = 0;
370                    // Status values:
371                    // 0 -> fetching environment variables, working directory and channels
372                    // 1 -> fetching the command and its arguments
373                    String dirString = null;
374                    File stdinFile = null;
375                    File stdoutFile = null;
376                    File stderrFile = null;
377                    ArrayList envsList = new ArrayList();
378                    ArrayList commandList = new ArrayList();
379                    for (int i = 0; i < size; i++) {
380                            String tk = (String) splitted.get(i);
381                            // Check the local status.
382                            if (status == 0) {
383                                    // Environment variables, working directory and channels
384                                    if (tk.startsWith("ENV:")) {
385                                            envsList.add(tk.substring(4));
386                                            continue;
387                                    } else if (tk.startsWith("DIR:")) {
388                                            dirString = tk.substring(4);
389                                            continue;
390                                    } else if (tk.startsWith("IN:")) {
391                                            stdinFile = new File(tk.substring(3));
392                                            continue;
393                                    } else if (tk.startsWith("OUT:")) {
394                                            stdoutFile = new File(tk.substring(4));
395                                            continue;
396                                    } else if (tk.startsWith("ERR:")) {
397                                            stderrFile = new File(tk.substring(4));
398                                            continue;
399                                    } else {
400                                            status = 1;
401                                    }
402                            }
403                            if (status == 1) {
404                                    // Command and its arguments
405                                    commandList.add(tk);
406                            }
407                    }
408                    // Command validation.
409                    String[] cmdarray = null;
410                    size = commandList.size();
411                    if (size == 0) {
412                            throw new Exception("Invalid cron line: " + line);
413                    } else {
414                            cmdarray = new String[size];
415                            for (int i = 0; i < size; i++) {
416                                    cmdarray[i] = (String) commandList.get(i);
417                            }
418                    }
419                    // Environments.
420                    String[] envs = null;
421                    size = envsList.size();
422                    if (size > 0) {
423                            envs = new String[size];
424                            for (int i = 0; i < size; i++) {
425                                    envs[i] = (String) envsList.get(i);
426                            }
427                    }
428                    // Working directory.
429                    File dir = null;
430                    if (dirString != null) {
431                            dir = new File(dirString);
432                            if (!dir.exists() || !dir.isDirectory()) {
433                                    throw new Exception(
434                                                    "Invalid cron working directory parameter at line: "
435                                                                    + line, new FileNotFoundException(dirString
436                                                                    + " doesn't exist or it is not a directory"));
437                            }
438                    }
439                    // Builds the task.
440                    ProcessTask task = new ProcessTask(cmdarray, envs, dir);
441                    // Channels.
442                    if (stdinFile != null) {
443                            task.setStdinFile(stdinFile);
444                    }
445                    if (stdoutFile != null) {
446                            task.setStdoutFile(stdoutFile);
447                    }
448                    if (stderrFile != null) {
449                            task.setStderrFile(stderrFile);
450                    }
451                    // End.
452                    table.add(new SchedulingPattern(pattern), task);
453            }

455            /**
456             * Escapes special chars occurrences.
457             * 
458             * @param str
459             *            The input stream.
460             * @return The decoded output stream.
461             */
462            private static String escape(String str) {
463                    int size = str.length();
464                    StringBuffer b = new StringBuffer();
465                    for (int i = 0; i < size; i++) {
466                            int skip = 0;
467                            char c = str.charAt(i);
468                            if (c == '\\') {
469                                    if (i < size - 1) {
470                                            char d = str.charAt(i + 1);
471                                            if (d == '"') {
472                                                    b.append('"');
473                                                    skip = 2;
474                                            } else if (d == '\\') {
475                                                    b.append('\\');
476                                                    skip = 2;
477                                            } else if (d == '/') {
478                                                    b.append('/');
479                                                    skip = 2;
480                                            } else if (d == 'b') {
481                                                    b.append('\b');
482                                                    skip = 2;
483                                            } else if (d == 'f') {
484                                                    b.append('\f');
485                                                    skip = 2;
486                                            } else if (d == 'n') {
487                                                    b.append('\n');
488                                                    skip = 2;
489                                            } else if (d == 'r') {
490                                                    b.append('\r');
491                                                    skip = 2;
492                                            } else if (d == 't') {
493                                                    b.append('\t');
494                                                    skip = 2;
495                                            } else if (d == 'u') {
496                                                    if (i < size - 5) {
497                                                            String hex = str.substring(i + 2, i + 6);
498                                                            try {
499                                                                    int code = Integer.parseInt(hex, 16);
500                                                                    if (code >= 0) {
501                                                                            b.append((char) code);
502                                                                            skip = 6;
503                                                                    }
504                                                            } catch (NumberFormatException e) {
505                                                                    ;
506                                                            }
507                                                    }
508                                            }
509                                    }
510                            }
511                            if (skip == 0) {
512                                    b.append(c);
513                            } else {
514                                    i += (skip - 1);
515                            }
516                    }
517                    return b.toString();
518            }
