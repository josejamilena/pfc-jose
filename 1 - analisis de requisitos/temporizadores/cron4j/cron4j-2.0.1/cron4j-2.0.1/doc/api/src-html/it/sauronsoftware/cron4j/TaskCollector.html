019    package it.sauronsoftware.cron4j;
020    
021    /**
022     * <p>
023     * This interface describes a task collector. Task collectors can be registered
024     * in a {@link Scheduler} instance with the
025     * {@link Scheduler#addTaskCollector(TaskCollector)} method. Any registered task
026     * collector is queried by the scheduler once a minute. The developer has to
027     * implement the {@link TaskCollector#getTasks()} method, returning a
028     * {@link TaskTable} whose elements has been collected with a custom logic. In
029     * example the list can be extracted from a database.
030     * </p>
031     * 
032     * @author Carlo Pelliccia
033     * @since 2.0
034     */
035    public interface TaskCollector {
036    
037            /**
038             * Once the instance has been registered on a {@link Scheduler} instance,
039             * with the {@link Scheduler#addTaskCollector(TaskCollector)} method, this
040             * method will be queried once a minute. It should return a custom
041             * {@link TaskTable} object. The scheduler instance will automatically
042             * iterate over the returned table elements, executing any task whose
043             * scheduling pattern is matching the current system time.
044             * 
045             * @return The task table that will be automatically injected in the
046             *         scheduler.
047             */
048            public TaskTable getTasks();
049    
050    }
