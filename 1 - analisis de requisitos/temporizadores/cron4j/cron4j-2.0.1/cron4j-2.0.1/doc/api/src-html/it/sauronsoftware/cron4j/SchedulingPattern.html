029    /**
030     * <p>
031     * A UNIX crontab-like pattern is a string splitted in five space separated
032     * parts. Each part is intented as:
033     * </p>
034     * <ol>
035     * <li><strong>Minutes sub-pattern</strong>. During which minutes of the hour
036     * should the task been launched? The values range is from 0 to 59.</li>
037     * <li><strong>Hours sub-pattern</strong>. During which hours of the day should
038     * the task been launched? The values range is from 0 to 23.</li>
039     * <li><strong>Days of month sub-pattern</strong>. During which days of the
040     * month should the task been launched? The values range is from 0 to 31.</li>
041     * <li><strong>Months sub-pattern</strong>. During which months of the year
042     * should the task been launched? The values range is from 1 (January) to 12
043     * (December), otherwise this sub-pattern allows the aliases &quot;jan&quot;,
044     * &quot;feb&quot;, &quot;mar&quot;, &quot;apr&quot;, &quot;may&quot;,
045     * &quot;jun&quot;, &quot;jul&quot;, &quot;aug&quot;, &quot;sep&quot;,
046     * &quot;oct&quot;, &quot;nov&quot; and &quot;dec&quot;.</li>
047     * <li><strong>Days of week sub-pattern</strong>. During which days of the week
048     * should the task been launched? The values range is from 0 (Sunday) to 6
049     * (Saturday), otherwise this sub-pattern allows the aliases &quot;sun&quot;,
050     * &quot;mon&quot;, &quot;tue&quot;, &quot;wed&quot;, &quot;thu&quot;,
051     * &quot;fri&quot; and &quot;sat&quot;.</li>
052     * </ol>
053     * <p>
054     * The star wildcard character is also admitted, indicating &quot;every minute
055     * of the hour&quot;, &quot;every hour of the day&quot;, &quot;every day of the
056     * month&quot;, &quot;every month of the year&quot; and &quot;every day of the
057     * week&quot;, according to the sub-pattern in which it is used.
058     * </p>
059     * <p>
060     * Once the scheduler is started, a task will be launched when the five parts in
061     * its scheduling pattern will be true at the same time.
062     * </p>
063     * <p>
064     * Some examples:
065     * </p>
066     * <p>
067     * <strong>5 * * * *</strong><br>
068     * This pattern causes a task to be launched once every hour, at the begin of
069     * the fifth minute (00:05, 01:05, 02:05 etc.).
070     * </p>
071     * <p>
072     * <strong>* * * * *</strong><br>
073     * This pattern causes a task to be launched every minute.
074     * </p>
075     * <p>
076     * <strong>* 12 * * Mon</strong><br>
077     * This pattern causes a task to be launched every minute during the 12th hour
078     * of Monday.
079     * </p>
080     * <p>
081     * <strong>* 12 16 * Mon</strong><br>
082     * This pattern causes a task to be launched every minute during the 12th hour
083     * of Monday, 16th, but only if the day is the 16th of the month.
084     * </p>
085     * <p>
086     * Every sub-pattern can contain two or more comma separated values.
087     * </p>
088     * <p>
089     * <strong>59 11 * * 1,2,3,4,5</strong><br>
090     * This pattern causes a task to be launched at 11:59AM on Monday, Tuesday,
091     * Wednesday, Thursday and Friday.
092     * </p>
093     * <p>
094     * Values intervals are admitted and defined using the minus character.
095     * </p>
096     * <p>
097     * <strong>59 11 * * 1-5</strong><br>
098     * This pattern is equivalent to the previous one.
099     * </p>
100     * <p>
101     * The slash character can be used to identify periodic values, in the form a/b.
102     * A sub-pattern with the slash character is satisfied when the value on the
103     * left divided by the one on the right gives an integer result (a % b == 0).
104     * </p>
105     * <p>
106     * <strong>*&#47;15 9-17 * * *</strong><br>
107     * This pattern causes a task to be launched every 15 minutes between the 9th
108     * and 17th hour of the day (9:00, 9:15, 9:30, 9:45 and so on... note that the
109     * last execution will be at 17:45).
110     * </p>
111     * <p>
112     * All the fresh described syntax rules can be used together.
113     * </p>
114     * <p>
115     * <strong>* 12 10-16/2 * *</strong><br>
116     * This pattern causes a task to be launched every minute during the 12th hour
117     * of the day, but only if the day is the 10th, the 12th, the 14th or the16th of
118     * the month.
119     * </p>
120     * <p>
121     * <strong>* 12 1-15,17,20-25 * *</strong><br>
122     * This pattern causes a task to be launched every minute during the 12th hour
123     * of the day, but the day of the month must be between the 1st and the 15th,
124     * the 20th and the 25, or at least it must be the 17th.
125     * </p>
126     * <p>
127     * Finally cron4j lets you combine more scheduling patterns into one, with the
128     * pipe character:
129     * </p>
130     * <p>
131     * <strong>0 5 * * *|8 10 * * *|22 17 * * *</strong><br>
132     * This pattern causes a task to be launched every day at 05:00, 10:08 and
133     * 17:22.
134     * </p>
135     * 
136     * @author Carlo Pelliccia
137     * @since 2.0
138     */

167             * Validates a string as a scheduling pattern.
168             * 
169             * @param schedulingPattern
170             *            The pattern to validate.
171             * @return true if the given string represents a valid scheduling pattern;
172             *         false otherwise.
173             */
174            public static boolean validate(String schedulingPattern) {
175                    try {
176                            new SchedulingPattern(schedulingPattern);
177                    } catch (InvalidPatternException e) {
178                            return false;
179                    }
180                    return true;
181            }

218            /**
219             * Builds a SchedulingPattern parsing it from a string.
220             * 
221             * @param pattern
222             *            The pattern as a crontab-like string.
223             * @throws InvalidPatternException
224             *             If the supplied string is not a valid pattern.
225             */
226            public SchedulingPattern(String pattern) throws InvalidPatternException {
227                    this.asString = pattern;
228                    StringTokenizer st1 = new StringTokenizer(pattern, "|");
229                    if (st1.countTokens() < 1) {
230                            throw new InvalidPatternException("invalid pattern: \"" + pattern + "\"");
231                    }
232                    while (st1.hasMoreTokens()) {
233                            String localPattern = st1.nextToken();
234                            StringTokenizer st2 = new StringTokenizer(localPattern, " \t");
235                            if (st2.countTokens() != 5) {
236                                    throw new InvalidPatternException("invalid pattern: \"" + localPattern + "\"");
237                            }
238                            try {
239                                    minuteMatchers.add(buildValueMatcher(st2.nextToken(), MINUTE_VALUE_PARSER));
240                            } catch (Exception e) {
241                                    throw new InvalidPatternException("invalid pattern \"" + localPattern + "\". Error parsing minutes field: " + e.getMessage() + ".");
242                            }
243                            try {
244                                    hourMatchers.add(buildValueMatcher(st2.nextToken(), HOUR_VALUE_PARSER));
245                            } catch (Exception e) {
246                                    throw new InvalidPatternException("invalid pattern \"" + localPattern + "\". Error parsing hours field: " + e.getMessage() + ".");
247                            }
248                            try {
249                                    dayOfMonthMatchers.add(buildValueMatcher(st2.nextToken(), DAY_OF_MONTH_VALUE_PARSER));
250                            } catch (Exception e) {
251                                    throw new InvalidPatternException("invalid pattern \"" + localPattern + "\". Error parsing days of month field: " + e.getMessage() + ".");
252                            }
253                            try {
254                                    monthMatchers.add(buildValueMatcher(st2.nextToken(), MONTH_VALUE_PARSER));
255                            } catch (Exception e) {
256                                    throw new InvalidPatternException("invalid pattern \"" + localPattern + "\". Error parsing months field: " + e.getMessage() + ".");
257                            }
258                            try {
259                                    dayOfWeekMatchers.add(buildValueMatcher(st2.nextToken(), DAY_OF_WEEK_VALUE_PARSER));
260                            } catch (Exception e) {
261                                    throw new InvalidPatternException("invalid pattern \"" + localPattern + "\". Error parsing days of week field: " + e.getMessage() + ".");
262                            }
263                            matcherSize++;
264                    }
265            }

267            /**
268             * A ValueMatcher utility builder.
269             * 
270             * @param str
271             *            The pattern part for the ValueMatcher creation.
272             * @param parser
273             *            The parser used to parse the values.
274             * @return The requested ValueMatcher.
275             * @throws Exception
276             *             If the supplied pattern part is not valid.
277             */
278            private ValueMatcher buildValueMatcher(String str, ValueParser parser) throws Exception {
279                    if (str.length() == 1 && str.equals("*")) {
280                            return new AlwaysTrueValueMatcher();
281                    }
282                    ArrayList values = new ArrayList();
283                    StringTokenizer st = new StringTokenizer(str, ",");
284                    while (st.hasMoreTokens()) {
285                            String element = st.nextToken();
286                            ArrayList local;
287                            try {
288                                    local = parseListElement(element, parser);
289                            } catch (Exception e) {
290                                    throw new Exception("invalid field \"" + str + "\", invalid element \"" + element + "\", " + e.getMessage());
291                            }
292                            for (Iterator i = local.iterator(); i.hasNext();) {
293                                    Object value = i.next();
294                                    if (!values.contains(value)) {
295                                            values.add(value);
296                                    }
297                            }
298                    }
299                    if (values.size() == 0) {
300                            throw new Exception("invalid field \"" + str + "\"");
301                    }
302                    return new IntArrayValueMatcher(values);
303            }

305            /**
306             * Parses an element of a list of values of the pattern.
307             * 
308             * @param str
309             *            The element string.
310             * @param parser
311             *            The parser used to parse the values.
312             * @return A list of integers representing the allowed values.
313             * @throws Exception
314             *             If the supplied pattern part is not valid.
315             */
316            private ArrayList parseListElement(String str, ValueParser parser) throws Exception {
317                    StringTokenizer st = new StringTokenizer(str, "/");
318                    int size = st.countTokens();
319                    if (size < 1 || size > 2) {
320                            throw new Exception("syntax error");
321                    }
322                    ArrayList values;
323                    try {
324                            values = parseRange(st.nextToken(), parser);
325                    } catch (Exception e) {
326                            throw new Exception("invalid range, " + e.getMessage());
327                    }
328                    if (size == 2) {
329                            String dStr = st.nextToken();
330                            int div;
331                            try {
332                                    div = Integer.parseInt(dStr);
333                            } catch (NumberFormatException e) {
334                                    throw new Exception("invalid divisor \"" + dStr + "\"");
335                            }
336                            if (div < 1) {
337                                    throw new Exception("non positive divisor \"" + div + "\"");
338                            }
339                            ArrayList values2 = new ArrayList();
340                            for (Iterator i = values.iterator(); i.hasNext();) {
341                                    Integer value = (Integer) i.next();
342                                    if (value.intValue() % div == 0) {
343                                            values2.add(value);
344                                    }
345                            }
346                            return values2;
347                    } else {
348                            return values;
349                    }
350            }

352            /**
353             * Parses a range of values.
354             * 
355             * @param str
356             *            The range string.
357             * @param parser
358             *            The parser used to parse the values.
359             * @return A list of integers representing the allowed values.
360             * @throws Exception
361             *             If the supplied pattern part is not valid.
362             */
363            private ArrayList parseRange(String str, ValueParser parser) throws Exception {
364                    if (str.equals("*")) {
365                            int[] all = parser.getAllPossibleValues();
366                            ArrayList values = new ArrayList();
367                            for (int i = 0; i < all.length; i++) {
368                                    values.add(new Integer(all[i]));
369                            }
370                            return values;
371                    }
372                    StringTokenizer st = new StringTokenizer(str, "-");
373                    int size = st.countTokens();
374                    if (size < 1 || size > 2) {
375                            throw new Exception("syntax error");
376                    }
377                    String v1Str = st.nextToken();
378                    int v1;
379                    try {
380                            v1 = parser.parse(v1Str);
381                    } catch (Exception e) {
382                            throw new Exception("invalid value \"" + v1Str + "\", " + e.getMessage());
383                    }
384                    if (size == 1) {
385                            ArrayList values = new ArrayList();
386                            values.add(new Integer(v1));
387                            return values;
388                    } else {
389                            String v2Str = st.nextToken();
390                            int v2;
391                            try {
392                                    v2 = parser.parse(v2Str);
393                            } catch (Exception e) {
394                                    throw new Exception("invalid value \"" + v2Str + "\", " + e.getMessage());
395                            }
396                            int s1 = Math.min(v1, v2);
397                            int s2 = Math.max(v1, v2);
398                            ArrayList values = new ArrayList();
399                            for (int i = s1; i <= s2; i++) {
400                                    values.add(new Integer(i));
401                            }
402                            return values;
403                    }
404            }

406            /**
407             * This methods returns true if the given timestamp (expressed as a UNIX-era
408             * millis value) matches the pattern, according to the given time zone.
409             * 
410             * @param timezone
411             *            A time zone.
412             * @param millis
413             *            The timestamp, as a UNIX-era millis value.
414             * @return true if the given timestamp matches the pattern.
415             */
416            public boolean match(TimeZone timezone, long millis) {
417                    GregorianCalendar gc = new GregorianCalendar();
418                    gc.setTime(new Date(millis));
419                    int minute = gc.get(Calendar.MINUTE);
420                    int hour = gc.get(Calendar.HOUR_OF_DAY);
421                    int dayOfMonth = gc.get(Calendar.DAY_OF_MONTH);
422                    int month = gc.get(Calendar.MONTH) + 1;
423                    int dayOfWeek = gc.get(Calendar.DAY_OF_WEEK) - 1;
424                    for (int i = 0; i < matcherSize; i++) {
425                            ValueMatcher minuteMatcher = (ValueMatcher) minuteMatchers.get(i);
426                            ValueMatcher hourMatcher = (ValueMatcher) hourMatchers.get(i);
427                            ValueMatcher dayOfMonthMatcher = (ValueMatcher) dayOfMonthMatchers.get(i);
428                            ValueMatcher monthMatcher = (ValueMatcher) monthMatchers.get(i);
429                            ValueMatcher dayOfWeekMatcher = (ValueMatcher) dayOfWeekMatchers.get(i);
430                            boolean eval = minuteMatcher.match(minute)
431                                            && hourMatcher.match(hour)
432                                            && dayOfMonthMatcher.match(dayOfMonth)
433                                            && monthMatcher.match(month)
434                                            && dayOfWeekMatcher.match(dayOfWeek);
435                            if (eval) {
436                                    return true;
437                            }
438                    }
439                    return false;
440            }

442            /**
443             * This methods returns true if the given timestamp (expressed as a UNIX-era
444             * millis value) matches the pattern, according to the system default time
445             * zone.
446             * 
447             * @param millis
448             *            The timestamp, as a UNIX-era millis value.
449             * @return true if the given timestamp matches the pattern.
450             */
451            public boolean match(long millis) {
452                    return match(TimeZone.getDefault(), millis);
453            }

464            /**
465             * This utility method changes an alias to an int value.
466             * 
467             * @param value
468             *            The value.
469             * @param aliases
470             *            The aliases list.
471             * @param offset
472             *            The offset appplied to the aliases list indices.
473             * @return The parsed value.
474             * @throws Exception
475             *             If the expressed values doesn't match any alias.
476             */
477            private static int parseAlias(String value, String[] aliases, int offset)
478                            throws Exception {
479                    for (int i = 0; i < aliases.length; i++) {
480                            if (aliases[i].equalsIgnoreCase(value)) {
481                                    return offset + i;
482                            }
483                    }
484                    throw new Exception("invalid alias \"" + value + "\"");
485            }

487            /**
488             * Definition for a value parser.
489             */
490            private static interface ValueParser {
491    
492                    /**
493                     * Attempts to parse a value.
494                     * 
495                     * @param value
496                     *            The value.
497                     * @return The parsed value.
498                     * @throws Exception
499                     *             If the value can't be parsed.
500                     */
501                    public int parse(String value) throws Exception;
502                    
503                    /**
504                     * Returns a list of all the values allowed.
505                     * 
506                     * @return A list of all the values allowed.
507                     */
508                    public int[] getAllPossibleValues();
509    
510            }

512            /**
513             * A simple value parser.
514             */
515            private static class SimpleValueParser implements ValueParser {
516    
517                    /**
518                     * The minimum allowed value.
519                     */
520                    protected int minValue;
521    
522                    /**
523                     * The maximum allowed value.
524                     */
525                    protected int maxValue;
526    
527                    /**
528                     * Builds the value parser.
529                     * 
530                     * @param minValue
531                     *            The minimum allowed value.
532                     * @param maxValue
533                     *            The maximum allowed value.
534                     */
535                    public SimpleValueParser(int minValue, int maxValue) {
536                            this.minValue = minValue;
537                            this.maxValue = maxValue;
538                    }
539    
540                    public int parse(String value) throws Exception {
541                            int i;
542                            try {
543                                    i = Integer.parseInt(value);
544                            } catch (NumberFormatException e) {
545                                    throw new Exception("invalid integer value");
546                            }
547                            if (i < minValue || i > maxValue) {
548                                    throw new Exception("value out of range");
549                            }
550                            return i;
551                    }
552    
553                    public int[] getAllPossibleValues() {
554                            int size = maxValue - minValue + 1;
555                            int[] ret = new int[size];
556                            for (int i = 0; i < size; i++) {
557                                    ret[i] = minValue + i;
558                            }
559                            return ret;
560                    }
561    
562            }

607             * The value parser for the months field.
608             */
609            private static class MonthValueParser extends SimpleValueParser {
610    
611                    /**
612                     * Months aliases.
613                     */
614                    private static String[] ALIASES = { "jan", "feb", "mar", "apr", "may", "jun", "jul", "aug", "sep", "oct", "nov", "dec" };
615    
616                    /**
617                     * Builds the months value parser.
618                     */
619                    public MonthValueParser() {
620                            super(1, 12);
621                    }
622    
623                    public int parse(String value) throws Exception {
624                            try {
625                                    // try as a simple value
626                                    return super.parse(value);
627                            } catch (Exception e) {
628                                    // try as an alias
629                                    return parseAlias(value, ALIASES, 1);
630                            }
631                    }
632                    
633            }

636             * The value parser for the months field.

640                    /**
641                     * Days of week aliases.
642                     */
643                    private static String[] ALIASES = { "sun", "mon", "tue", "wed", "thu", "fri", "sat" };
644    
645                    /**
646                     * Builds the months value parser.
647                     */
648                    public DayOfWeekValueParser() {
649                            super(0, 7);
650                    }
651    
652                    public int parse(String value) throws Exception {
653                            try {
654                                    // try as a simple value
655                                    return super.parse(value) % 7;
656                            } catch (Exception e) {
657                                    // try as an alias
658                                    return parseAlias(value, ALIASES, 0);
